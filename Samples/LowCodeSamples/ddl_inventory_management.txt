
-- 仕入先
CREATE TABLE supplier (
    id SERIAL PRIMARY KEY,
    name TEXT,
    contact_name TEXT,
    phone TEXT,
    email TEXT,
    address TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 倉庫
CREATE TABLE warehouse (
    id SERIAL PRIMARY KEY,
    name TEXT,
    address TEXT,
    note TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- 商品
CREATE TABLE product (
    id SERIAL PRIMARY KEY,
    code TEXT,
    name TEXT,
    category TEXT,
    unit TEXT,
    unit_price INTEGER,
    reorder_point INTEGER,
    safety_stock INTEGER,
    supplier_id INTEGER,
    note TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE UNIQUE INDEX idx_product_code_unique ON product(code);
CREATE INDEX idx_product_supplier ON product(supplier_id);
CREATE INDEX idx_product_category ON product(category);

-- 入庫ヘッダ
CREATE TABLE receiving (
    id SERIAL PRIMARY KEY,
    receiving_date DATE,
    warehouse_id INTEGER,
    supplier_id INTEGER,
    note TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_receiving_warehouse ON receiving(warehouse_id);
CREATE INDEX idx_receiving_supplier ON receiving(supplier_id);
CREATE INDEX idx_receiving_date ON receiving(receiving_date);

-- 入庫明細
CREATE TABLE receiving_detail (
    id SERIAL PRIMARY KEY,
    receiving_id INTEGER,
    product_id INTEGER,
    product_code TEXT,
    product_name TEXT,
    quantity INTEGER,
    unit_price INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_receiving_detail_receiving ON receiving_detail(receiving_id);
CREATE INDEX idx_receiving_detail_product ON receiving_detail(product_id);

-- 出庫ヘッダ
CREATE TABLE shipping (
    id SERIAL PRIMARY KEY,
    shipping_date DATE,
    warehouse_id INTEGER,
    shipped_to TEXT,
    note TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_shipping_warehouse ON shipping(warehouse_id);
CREATE INDEX idx_shipping_date ON shipping(shipping_date);

-- 出庫明細
CREATE TABLE shipping_detail (
    id SERIAL PRIMARY KEY,
    shipping_id INTEGER,
    product_id INTEGER,
    product_code TEXT,
    product_name TEXT,
    quantity INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_shipping_detail_shipping ON shipping_detail(shipping_id);
CREATE INDEX idx_shipping_detail_product ON shipping_detail(product_id);

-- 棚卸しヘッダ
CREATE TABLE stocktaking (
    id SERIAL PRIMARY KEY,
    stocktake_date DATE,
    warehouse_id INTEGER,
    note TEXT,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_stocktaking_warehouse ON stocktaking(warehouse_id);
CREATE INDEX idx_stocktaking_date ON stocktaking(stocktake_date);

-- 棚卸し明細
CREATE TABLE stocktaking_detail (
    id SERIAL PRIMARY KEY,
    stocktaking_id INTEGER,
    product_id INTEGER,
    product_code TEXT,
    product_name TEXT,
    inventory_id INTEGER,
    book_quantity INTEGER,
    actual_quantity INTEGER,
    difference INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_stocktaking_detail_stocktaking ON stocktaking_detail(stocktaking_id);
CREATE INDEX idx_stocktaking_detail_product ON stocktaking_detail(product_id);
CREATE INDEX idx_stocktaking_detail_inventory ON stocktaking_detail(inventory_id);

-- 在庫
CREATE TABLE inventory (
    id SERIAL PRIMARY KEY,
    product_id INTEGER,
    warehouse_id INTEGER,
    current_stock INTEGER,
    version INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP,
    UNIQUE (product_id, warehouse_id)
);
CREATE INDEX idx_inventory_product ON inventory(product_id);
CREATE INDEX idx_inventory_warehouse ON inventory(warehouse_id);

-- 発注ヘッダ
CREATE TABLE purchase_order (
    id SERIAL PRIMARY KEY,
    order_no TEXT,
    order_date DATE,
    desired_delivery_date DATE,
    supplier_id INTEGER,
    warehouse_id INTEGER,
    status TEXT,
    note TEXT,
    version INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE UNIQUE INDEX idx_purchase_order_no_unique ON purchase_order(order_no) WHERE order_no IS NOT NULL;
CREATE INDEX idx_purchase_order_supplier ON purchase_order(supplier_id);
CREATE INDEX idx_purchase_order_warehouse ON purchase_order(warehouse_id);
CREATE INDEX idx_purchase_order_status ON purchase_order(status);
CREATE INDEX idx_purchase_order_date ON purchase_order(order_date);

-- 発注明細
CREATE TABLE purchase_order_line (
    id SERIAL PRIMARY KEY,
    purchase_order_id INTEGER,
    product_id INTEGER,
    product_code TEXT,
    product_name TEXT,
    quantity INTEGER,
    unit_price INTEGER,
    received_quantity INTEGER,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);
CREATE INDEX idx_purchase_order_line_order ON purchase_order_line(purchase_order_id);
CREATE INDEX idx_purchase_order_line_product ON purchase_order_line(product_id);

-- 在庫スナップショット（入出庫実績から集計して初期化）
INSERT INTO inventory (product_id, warehouse_id, current_stock, version, created_at, updated_at)
SELECT
    p.id,
    w.id,
    COALESCE(ri.in_qty, 0) - COALESCE(si.out_qty, 0) + COALESCE(ai.adj_qty, 0),
    0,
    '2026-05-12 00:00:00',
    '2026-05-12 00:00:00'
FROM product p
CROSS JOIN warehouse w
LEFT JOIN (
    SELECT r.warehouse_id, rd.product_id, SUM(rd.quantity) AS in_qty
    FROM receiving r JOIN receiving_detail rd ON rd.receiving_id = r.id
    GROUP BY r.warehouse_id, rd.product_id
) ri ON ri.warehouse_id = w.id AND ri.product_id = p.id
LEFT JOIN (
    SELECT s.warehouse_id, sd.product_id, SUM(sd.quantity) AS out_qty
    FROM shipping s JOIN shipping_detail sd ON sd.shipping_id = s.id
    GROUP BY s.warehouse_id, sd.product_id
) si ON si.warehouse_id = w.id AND si.product_id = p.id
LEFT JOIN (
    SELECT st.warehouse_id, std.product_id, SUM(std.difference) AS adj_qty
    FROM stocktaking st JOIN stocktaking_detail std ON std.stocktaking_id = st.id
    GROUP BY st.warehouse_id, std.product_id
) ai ON ai.warehouse_id = w.id AND ai.product_id = p.id
WHERE ri.in_qty IS NOT NULL OR si.out_qty IS NOT NULL OR ai.adj_qty IS NOT NULL;
