CREATE TABLE department (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT,
    created_at TEXT,
    updated_at TEXT
);

CREATE TABLE members (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT,
    email TEXT,
    role TEXT,
    department_id INTEGER,
    created_at TEXT,
    updated_at TEXT
);

CREATE TABLE project (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT,
    description TEXT,
    start_date TEXT,
    end_date TEXT,
    status TEXT,
    owner_member_id INTEGER,
    created_at TEXT,
    updated_at TEXT
);
CREATE INDEX idx_project_owner ON project(owner_member_id);

CREATE TABLE project_member (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER,
    member_id INTEGER,
    project_role TEXT,
    joined_date TEXT,
    created_at TEXT,
    updated_at TEXT
);
CREATE INDEX idx_project_member_project ON project_member(project_id);
CREATE INDEX idx_project_member_member ON project_member(member_id);

CREATE TABLE task (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    project_id INTEGER,
    name TEXT,
    description TEXT,
    assignee_member_id INTEGER,
    start_date TEXT,
    end_date TEXT,
    status TEXT,
    priority TEXT,
    progress_rate INTEGER,
    sort_order INTEGER,
    created_at TEXT,
    updated_at TEXT
);
CREATE INDEX idx_task_project ON task(project_id);
CREATE INDEX idx_task_assignee ON task(assignee_member_id);
CREATE INDEX idx_task_status ON task(status);
